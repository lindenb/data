\section{Unix}
\subsection{Regex with one column}
\begin{verbatim}
$ echo -e "A B C\nA T G C\n" | awk -F ' ' '($2 ~ "B" )'
A B C
\end{verbatim}
\subsection{Replace one column}
\begin{verbatim}
$ echo -e "A B C\nA T G C\n" | awk -F ' ' '($3 = "B" )'
A B B
A T B C
  B

\end{verbatim}
